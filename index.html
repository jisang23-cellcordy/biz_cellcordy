        function wrapLabel(label, maxChars = 16) {

            const words = label.split(' ');

            const lines = [];

            let currentLine = '';

           

            words.forEach(word => {

                if ((currentLine + word).length > maxChars) {

                    if (currentLine) lines.push(currentLine.trim());

                    currentLine = word + ' ';

                } else {

                    currentLine += word + ' ';

                }

            });

            if (currentLine) lines.push(currentLine.trim());

            return lines;

        }
